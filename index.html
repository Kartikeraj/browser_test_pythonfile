PyScript Demo — Number * 2

from js import document def run_python(evt=None): v = document.getElementById("user-input").value if v == "": document.getElementById("output").innerHTML = "Please enter a number!" return try: n = float(v) except Exception as e: document.getElementById("output").innerHTML = f"Not a valid number: {e}" return document.getElementById("output").innerHTML = f"Result: {n * 2}" # attach handler document.getElementById("run-btn").addEventListener("click", run_python)
