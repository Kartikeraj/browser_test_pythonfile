Love Score Calculator ❤️

# ---------- Python (runs in browser) ---------- from js import document, window from pyodide.ffi import create_once_callable def calculate_love_score(name1, name2): name3 = name1 + name2 count1 = sum(1 for i in name3.lower() if i in "true") count2 = sum(1 for i in name3.lower() if i in "love") # combine digits (e.g. 3 and 4 -> 34) result = int(str(count1) + str(count2)) return result def py_handler(evt): # fetch inputs n1 = document.getElementById("name1").value.strip() n2 = document.getElementById("name2").value.strip() if not n1 or not n2: document.getElementById("output").innerHTML = "Please enter both names." return score = calculate_love_score(n1, n2) document.getElementById("output").innerHTML = f"Love Score: {score}% ❤️" # Create a JS-callable that will live in JS land and won't be auto-destroyed. # Storing it on window ensures it persists. js_callable = create_once_callable(py_handler) window.love_calc_handler = js_callable # Attach the JS-callable to the button's click event. document.getElementById("calc-btn").addEventListener("click", window.love_calc_handler)
