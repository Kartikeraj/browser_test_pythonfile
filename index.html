Love Score Calculator ❤️

from js import document, window from pyodide.ffi import create_proxy # PERSISTENT callback def calculate_love_score(name1, name2): name3 = name1 + name2 count1 = sum(1 for i in name3.lower() if i in "true") count2 = sum(1 for i in name3.lower() if i in "love") return int(str(count1) + str(count2)) def handler(event): n1 = document.getElementById("name1").value.strip() n2 = document.getElementById("name2").value.strip() if not n1 or not n2: document.getElementById("output").innerHTML = "Please enter both names." return score = calculate_love_score(n1, n2) document.getElementById("output").innerHTML = f"Love Score: {score}% ❤️" # Persistent proxy that will NEVER auto-destroy proxy = create_proxy(handler) # Store on window so GC won’t remove it window.love_calc_handler = proxy # Attach once and works forever document.getElementById("calc-btn").addEventListener("click", window.love_calc_handler)
